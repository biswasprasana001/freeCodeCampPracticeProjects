function titleCase(str) {
  let updatedWords = [];
  let words = str.toLowerCase().split(' ');
  // console.log(words);
  for(let i = 0; i < words.length; i++) {
    let firstLtr = words[i].charAt(0);
    // console.log(firstLtr);
    let upperCaseFirstLtr = firstLtr.toUpperCase();
    // console.log(upperCaseFirstLtr);
    updatedWords.push(words[i].replace(firstLtr, upperCaseFirstLtr));
}
// console.log(updatedWords.join(' '));
return updatedWords.join(' ');
}

titleCase("I'm a little tea pot");
